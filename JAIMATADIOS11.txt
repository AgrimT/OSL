#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, head, i, j, min, pos, total = 0;
    printf("Enter number of requests: ");
    scanf("%d", &n);
    int rq[n], done[n];
    printf("Enter requests: ");
    for(i = 0; i < n; i++){ scanf("%d", &rq[i]); done[i] = 0; }
    printf("Enter head position: ");
    scanf("%d", &head);

    for(i = 0; i < n; i++){
        min = 9999;
        for(j = 0; j < n; j++){
            int diff = abs(head - rq[j]);
            if(!done[j] && diff < min){ min = diff; pos = j; }
        }
        total += min;
        head = rq[pos];
        done[pos] = 1;
        printf("Move to %d\n", rq[pos]);
    }
    printf("Total Seek Time = %d\n", total);
    printf("Average Seek Time = %.2f\n", (float)total/n);
    return 0;
}
