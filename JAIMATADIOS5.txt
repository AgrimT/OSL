#include <stdio.h>

int main() {
    int n, tq, i, time = 0, done = 0;
    printf("Enter number of processes: ");
    scanf("%d", &n);
    int at[n], bt[n], rt[n];
    for (i = 0; i < n; i++) {
        printf("Enter Arrival and Burst time of P%d: ", i+1);
        scanf("%d%d", &at[i], &bt[i]);
        rt[i] = bt[i];
    }
    printf("Enter Time Quantum: ");
    scanf("%d", &tq);

    printf("\nGantt Chart: ");
    while (done < n) {
        int flag = 0;
        for (i = 0; i < n; i++) {
            if (at[i] <= time && rt[i] > 0) {
                flag = 1;
                printf("P%d ", i+1);
                if (rt[i] > tq) {
                    time += tq;
                    rt[i] -= tq;
                } else {
                    time += rt[i];
                    rt[i] = 0;
                    done++;
                }
            }
        }
        if (!flag) time++;
    }
    printf("\n");
    return 0;
}
