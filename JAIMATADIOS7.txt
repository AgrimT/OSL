#include <stdio.h>
#include <pthread.h>
#include <semaphore.h>
#include <unistd.h>

sem_t wrt;
pthread_mutex_t mutex;
int readcount = 0;

void *reader(void *rno) {
    pthread_mutex_lock(&mutex);
    readcount++;
    if (readcount == 1) sem_wait(&wrt);
    pthread_mutex_unlock(&mutex);

    printf("Reader %d reading\n", *(int*)rno);
    sleep(1);

    pthread_mutex_lock(&mutex);
    readcount--;
    if (readcount == 0) sem_post(&wrt);
    pthread_mutex_unlock(&mutex);
    printf("Reader %d done\n", *(int*)rno);
    return NULL;
}

void *writer(void *wno) {
    sem_wait(&wrt);
    printf("Writer %d writing\n", *(int*)wno);
    sleep(1);
    printf("Writer %d done\n", *(int*)wno);
    sem_post(&wrt);
    return NULL;
}

int main() {
    int i, r, w, id[10];
    pthread_t R[10], W[10];
    pthread_mutex_init(&mutex, 0);
    sem_init(&wrt, 0, 1);

    printf("Readers and Writers: ");
    scanf("%d%d", &r, &w);

    for (i = 0; i < r; i++) { id[i] = i + 1; pthread_create(&R[i], 0, reader, &id[i]); }
    for (i = 0; i < w; i++) { id[i] = i + 1; pthread_create(&W[i], 0, writer, &id[i]); }

    for (i = 0; i < r; i++) pthread_join(R[i], 0);
    for (i = 0; i < w; i++) pthread_join(W[i], 0);

    pthread_mutex_destroy(&mutex);
    sem_destroy(&wrt);
    return 0;
}
