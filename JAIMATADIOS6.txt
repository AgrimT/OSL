#include <stdio.h>
#include <pthread.h>
#include <semaphore.h>
#include <stdlib.h>

#define SIZE 5

int buffer[SIZE], in = 0, out = 0;
sem_t empty, full;
pthread_mutex_t mutex;

void *producer(void *pno) {
    for (int i = 0; i < SIZE; i++) {
        int item = rand() % 100;
        sem_wait(&empty);
        pthread_mutex_lock(&mutex);
        buffer[in] = item;
        printf("Producer %d -> Item %d\n", *(int*)pno, item);
        in = (in + 1) % SIZE;
        pthread_mutex_unlock(&mutex);
        sem_post(&full);
    }
    return NULL;
}

void *consumer(void *cno) {
    for (int i = 0; i < SIZE; i++) {
        sem_wait(&full);
        pthread_mutex_lock(&mutex);
        int item = buffer[out];
        printf("Consumer %d <- Item %d\n", *(int*)cno, item);
        out = (out + 1) % SIZE;
        pthread_mutex_unlock(&mutex);
        sem_post(&empty);
    }
    return NULL;
}

int main() {
    pthread_t p[2], c[2];
    int id[] = {1, 2};
    pthread_mutex_init(&mutex, 0);
    sem_init(&empty, 0, SIZE);
    sem_init(&full, 0, 0);

    pthread_create(&p[0], 0, producer, &id[0]);
    pthread_create(&c[0], 0, consumer, &id[0]);
    pthread_create(&p[1], 0, producer, &id[1]);
    pthread_create(&c[1], 0, consumer, &id[1]);

    for (int i = 0; i < 2; i++) {
        pthread_join(p[i], 0);
        pthread_join(c[i], 0);
    }

    pthread_mutex_destroy(&mutex);
    sem_destroy(&empty);
    sem_destroy(&full);
    return 0;
}
